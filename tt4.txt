    {

        protected void InsertedCmd(object sender, FormViewInsertedEventArgs e)
        {
            if (e.Exception != null)
            {
                e.KeepInInsertMode = true;
                e.ExceptionHandled = true;
                errmsg.Text = "Lỗi khi nhập : " + e.Exception.Message;
